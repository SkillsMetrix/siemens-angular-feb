import { Directive, ElementRef, Renderer2 } from '@angular/core';

@Directive({
  selector: '[appColor]',
  standalone: true
})
export class ColorDirective {
  private color='plum'
  constructor(private er:ElementRef,private ren:Renderer2){}
     ngOnInit(): void {
         this.ren.setStyle(this.er.nativeElement,'color',this.color)
     }

}




import { Directive, ElementRef, OnInit, Renderer2 } from '@angular/core';

@Directive({
  selector: '[appFontWt]',
  standalone: true
})
export class FontWtDirective implements OnInit {
private weight='bold'
constructor(private er:ElementRef,private ren:Renderer2){}
   ngOnInit(): void {
       this.ren.setStyle(this.er.nativeElement,'font-weight',this.weight)
   }
}
